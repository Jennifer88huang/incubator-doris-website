# group_concat

# description

# Syntax

VARCHAR group_concat(VARCHAR str[, VARCHAR sep])

sum() group_concat sep group by

# example

mysql> select value from test;
+-------+
| value |
+-------+
| a     |
| b     |
| c     |
+-------+

mysql> select group_concat(value) from test;
+-----------------------+
| group_concat(`value`) |
+-----------------------+
| a, b, c               |
+-----------------------+

mysql> select group_concat(value, " ") from test;
+----------------------------+
| group_concat(`value`, ' ') |
+----------------------------+
| a b c                      |
+----------------------------+

# keyword

GROUP_CONCAT,GROUP,CONCAT
